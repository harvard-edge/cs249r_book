% =============================================================================
% HALF-TITLE PAGE (Volume I)
% =============================================================================
% This file defines \halftitle, which is called from inside the redefined
% \frontmatter macro in header-includes.tex. The Quarto titlepage extension
% calls \begin{frontmatter} at the start of \begin{document}, which expands to
% the redefined \frontmatter that emits the half-title here BEFORE the
% titlepage extension's title page block runs.
%
% Standard academic book sequence:
%   half-title (page i)
%   blank verso (page ii)
%   title page proper (page iii) -- rendered by Quarto titlepage extension
%   ...
%
% The half-title shows only the book title -- no author, no publisher, no date.
%
% Color note: \color{crimson} resolves via the alias chain in
% header-includes.tex (\colorlet{crimson}{accentcolor}). For vol1 production,
% accentcolor is Harvard crimson; for the copyedit PDF, theme-colors-vol1-
% copyedit.tex sets accentcolor to black, which cascades here automatically.
%
% IMPORTANT: This file must be loaded via include-in-header (preamble) so the
% \renewcommand below registers BEFORE \begin{document} fires. Loading via
% include-before-body would be too late.
% =============================================================================

\renewcommand{\halftitle}{%
  \thispagestyle{empty}%
  \begin{center}%
  \vspace*{0.3\textheight}%
  {\fontsize{24pt}{28pt}\selectfont\bfseries\color{crimson} Introduction to\\[0.4em] Machine Learning Systems}\\[2em]%
  {\large\itshape Volume~I}%
  \vfill%
  \end{center}%
  \clearpage%
  \thispagestyle{empty}\null\clearpage%  Blank verso (back of half-title)
}
