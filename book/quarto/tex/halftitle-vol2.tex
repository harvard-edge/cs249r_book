% HALF-TITLE PAGE (Volume II)
% =============================================================================
% This file defines \halftitle, which is called from inside the Quarto
% titlepage extension at the seam between the coverpage block and the
% titlepage block (see _extensions/mlsysbook-ext/titlepage/before-body.tex).
% The effect is the standard academic book front-matter sequence:
%
%   cover art    (page 1, rendered by Quarto coverpage)
%   blank verso  (page 2)
%   half-title   (page 3, rendered by \halftitle defined here)
%   blank verso  (page 4)
%   title page   (page 5, rendered by Quarto titlepage)
%
% The half-title shows only the book title -- no author, no publisher, no date.
%
% Color note: \color{crimson} resolves via the alias chain in
% header-includes.tex (\colorlet{crimson}{accentcolor}). For vol2 production,
% accentcolor is ETH Zurich Blue; for the copyedit PDF, theme-colors-vol2-
% copyedit.tex sets accentcolor to black, which cascades here automatically.
%
% IMPORTANT: This file must be loaded via include-in-header (preamble) so the
% \renewcommand below registers BEFORE \begin{document} fires. Loading via
% include-before-body would be too late.
% =============================================================================

\renewcommand{\halftitle}{%
  \thispagestyle{empty}%
  \begin{center}%
  \vspace*{0.3\textheight}%
  {\fontsize{24pt}{28pt}\selectfont\bfseries\color{crimson} Machine Learning Systems\\[0.4em] at Scale}\\[2em]%
  {\large\itshape Volume~II}%
  \vfill%
  \end{center}%
  \clearpage%
  \thispagestyle{empty}\null\clearpage%  Blank verso (back of half-title)
}
