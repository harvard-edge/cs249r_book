\section{Iron Law of Workflow}
% =============================================================================

\mlsysfocus{The Iron Law of Workflow}{%
Each lifecycle stage maps to a specific term in:\\[0.3cm]
$T_{\text{total}} = \underbrace{\dfrac{D_{\text{vol}}}{BW}}_{\text{Data Collection}} +
\underbrace{\dfrac{O}{R_{\text{peak}} \cdot \eta}}_{\text{Model Dev}} +
\underbrace{L_{\text{lat}}}_{\text{Deployment}}$%
}

\begin{frame}{Iron Law of Workflow: Stages Map to Terms}
\note{
% -- LINK: Students know the Iron Law from Ch1 and the lifecycle stages from
% this chapter. This slide bridges them: each stage optimizes a specific term.
%
% -- NARRATE: ``Each lifecycle stage maps to a specific Iron Law term. Data
% Collection determines $D_{\text{vol}}$---how much data and how fast it
% moves. Model Development defines $O$---how many operations the model
% requires. Deployment minimizes $L_{\text{lat}}$---the orchestration overhead.
% Workflow management is mathematically equivalent to minimizing the Iron Law.
% If you only optimize the model (compute term) but neglect data collection
% (data term), you are optimizing the wrong term.''
%
% -- ENGAGE: ``If you only optimize the model (compute term), what happens
% to the system?'' Give 10 seconds.
% Expected: data term or overhead term dominates---you improved the wrong
% thing. This is the Bottleneck Principle from Ch2 applied to the workflow.
%
% -- WARN: Students optimize their favorite stage (usually model development)
% and neglect the others. Correct framing: the slowest stage dominates the
% workflow, just as the slowest Iron Law term dominates latency.
%
% -- FLEX: [CORE] Bridges the Iron Law to workflow management.
% IF SHORT: State the mapping in one sentence and move on.
}

\begin{frame}{Constraint Propagation Principle}
\note{

\begin{frame}{Emergent Complexity}

\begin{frame}{Fallacies}
\note{
% -- LINK: The chapter covered the lifecycle, workflow differences, constraint
% propagation, and systems thinking. These fallacies test internalization.
%
% -- NARRATE: ``Four misconceptions. First: `ML can follow traditional
% waterfall'---60--80\% of ML projects fail, and forcing waterfall misses the
% 4--8 iteration cycles production requires. Second: `passing evaluation =
% ready'---the DR model had AUC 0.99 in the lab but 78\% in the field; a
% Stage-5 discovery costs $16\times$. Third: `more data always helps'---
% doubling from 500K to 1M often improves accuracy less than 1\% while
% doubling costs. Fourth: `data prep is one-time'---models degrade 5--10\%
% within months; emergency retraining costs 3--5$\times$ proactive monitoring.''
%
% -- FLEX: [CORE] Corrects the most common misconceptions.
% IF SHORT: Focus on the first two fallacies.
}

\footnotesize
\textbf{Fallacy:} \textit{ML development can follow traditional software workflows.}\\
{\scriptsize 60--80\% of ML projects never reach production. Projects forced into waterfall miss the 4--8 iteration cycles production requires.}

\vspace{0.05cm}
\textbf{Fallacy:} \textit{Passing model evaluation means ready for deployment.}\\
{\scriptsize DR model: AUC 0.99 in lab, sensitivity 78\% in rural clinic. Stage-5 discovery costs $16\times$.}

\vspace{0.05cm}
\textbf{Fallacy:} \textit{More data always improves performance.}\\
{\scriptsize Doubling from 500K to 1M often improves accuracy <1\% while doubling costs. 100K clean beats 1M noisy.}

\vspace{0.05cm}
\textbf{Fallacy:} \textit{Data preparation is a one-time step.}\\
{\scriptsize Models degrade 5--10\% within months. Emergency retraining costs 3--5$\times$ proactive monitoring.}

\end{frame}

\begin{frame}{Pitfalls}
\note{
% -- LINK: Fallacies named misconceptions. Pitfalls name operational mistakes.
%
% -- NARRATE: ``Three pitfalls. First: skipping shadow mode validation saves
% 2 weeks but causes 10--50$\times$ latency spikes and 6--8 weeks of emergency
% remediation. Second: deferring deployment paradigm selection until after
% model development---exactly the Day-150 Disaster. A team develops a 2 GB
% ensemble, then discovers the TinyML target has 256 KB. Constraint
% Propagation: $2^4 = 16\times$ rework. Third: treating the model as the
% sole deliverable---without weights + code + environment + config bundled
% together, `works on my machine' means broken in production.''
%
% -- FLEX: [CORE] Operational lessons.
% IF SHORT: Cover the first two pitfalls---they connect directly to the
% chapter's central themes.
}

\footnotesize
\textbf{Pitfall:} \textit{Skipping validation to accelerate timelines.}\\
{\scriptsize Skipping shadow mode causes 10--50$\times$ latency spikes. ``Saving'' 2 weeks costs 6--8 weeks of emergency remediation.}

\vspace{0.08cm}
\textbf{Pitfall:} \textit{Deferring deployment paradigm selection until after model development.}\\
{\scriptsize A team develops a 2 GB ensemble, discovers TinyML target with 256 KB. Constraint Propagation: $2^4 = 16\times$ rework.}

\vspace{0.08cm}
\textbf{Pitfall:} \textit{Treating the model as the sole deliverable.}\\
{\scriptsize Reproducible artifact = weights + code + environment + config. Without bundling, ``works on my machine'' = broken.}

\end{frame}

\begin{frame}{Key Takeaways}
\note{
% -- NARRATE: ``Reveal time. Walk through each bullet. Emphasize the three
% quantitative anchors: 60--80\% data time, 4--8 iteration cycles needed,
% $2^{N-1}$ cost escalation. The Day-150 Disaster is the visceral story;

%
% -- FLEX: [CORE] Summary slide.
% IF SHORT: Hit the dual pipelines, constraint propagation, and silent

\footnotesize
\begin{itemize}\setlength\itemsep{1pt}
  \item \textbf{Two pipelines, one system}: Data and model development run in parallel, unified by continuous feedback loops.
  \item \textbf{60--80\% of effort is data-related}: Model development is only 10--20\%. Plan accordingly.
  \item \textbf{Iteration velocity compounds}: 10 experiments/day beats 1/week, even starting 5\% behind (Iteration Tax).
  \item \textbf{Constraint propagation is exponential}: $\text{Cost} = 2^{(N-1)}$. A Stage-5 discovery costs $16\times$ base effort.
  \item \textbf{Each stage maps to an Iron Law term}: Data $\to D_{\text{vol}}$, Model $\to O$, Deploy $\to L_{\text{lat}}$.
  \item \textbf{ML systems degrade silently}: Data drift, not code bugs. Continuous monitoring from day one.
  \item \textbf{Systems thinking}: Constraint propagation + multi-scale feedback + emergent complexity.
\end{itemize}
